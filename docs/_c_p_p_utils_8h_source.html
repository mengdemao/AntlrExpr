21 // Using RAII + a lambda to implement a "finally" replacement.
22 template <typename OnEnd>
23 struct FinalAction {
24 FinalAction(OnEnd f) : _cleanUp { std::move(f) } {}
25 FinalAction(FinalAction &&other) :
26 _cleanUp(std::move(other._cleanUp)), _enabled(other._enabled) {
27 other._enabled = false; // Don't trigger the lambda after ownership has moved.
28 }
29 ~FinalAction() { if (_enabled) _cleanUp(); }
30
31 void disable() { _enabled = false; }
32 private:
33 OnEnd _cleanUp;
34 bool _enabled {true};
35 };
36
37 template <typename OnEnd>
38 FinalAction<OnEnd> finally(OnEnd f) {
39 return FinalAction<OnEnd>(std::move(f));
40 }
